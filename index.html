// ============================================================ // PHASE 2: DEEP RECURSIVE TIME-SERIES AGGREGATOR // Scans Sub-Folders (Years -> Months) & Builds Cartesian Grid // ============================================================ function compileHistoricalPrices() { const folderId = "14eKJhVQX_eK2xMG2-QW0KbGKmpHqsJMQ"; const targetSheetName = "Hist. Prices"; const targetHeaders = [ "DATE", "ISIN", "COMPANY_NAME", "STATUS", "VWAP", "PREVIOUS", "Change", "VOLUME", "Turnover", "INTRADAY_HIGH", "INTRADAY_LOW", "52WK_HIGH", "52WK_LOW", "BID", "ASK" ]; const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(targetSheetName); if (!sheet) sheet = ss.insertSheet(targetSheetName); sheet.clear(); sheet.getRange(1, 1, 1, targetHeaders.length).setValues([targetHeaders]) .setBackground("#1a1a2e").setFontColor("#ffffff").setFontWeight("bold").setFontSize(10); sheet.setFrozenRows(1); // 🛡️ Retry Engine for the Root Folder let rootFolder; let retryCount = 0; while (retryCount < 5) { try { rootFolder = DriveApp.getFolderById(folderId); break; } catch (e) { retryCount++; Utilities.sleep(2000 * retryCount); if (retryCount >= 5) { console.error("Google Drive API is currently down. Please try again in a few minutes."); return; } } } const dataMap = {}; const uniqueDates = new Set(); const companyMemory = {}; let processedFiles = 0; console.log("🚀 Initiating Deep Recursive Scan of Sub-Folders..."); // ── RECURSIVE SCANNER FUNCTION ── function processFolder(folder) { // 1. Process files in the current folder const files = folder.getFiles(); while (files.hasNext()) { const file = files.next(); const fileName = file.getName(); const mimeType = file.getMimeType(); let rawData2D = []; if (mimeType === MimeType.GOOGLE_SHEETS) { try { const tempSS = SpreadsheetApp.openById(file.getId()); rawData2D = tempSS.getSheets()[0].getDataRange().getValues(); } catch(e) { continue; } } else if (mimeType === MimeType.CSV || fileName.toLowerCase().endsWith('.csv') || mimeType === 'text/plain') { try { const csvText = file.getBlob().getDataAsString(); if (csvText) rawData2D = Utilities.parseCsv(csvText); } catch(e) { continue; } } // If it is an Excel file, we must skip it (Google Apps Script cannot parse .xlsx directly to arrays) else if (mimeType.includes("excel") || mimeType.includes("spreadsheetml")) { console.warn(`⚠️ Skipped Excel File: ${fileName}. Please convert .xlsx to Google Sheets or CSV.`); continue; } else { continue; } if (rawData2D.length < 2) continue; processedFiles++; // Omni-Header Mapper (Accounts for 2008-2026 structural changes) // Look for the header row. In older files, headers might not be on row 0. let headerRowIdx = 0; for (let r = 0; r < Math.min(10, rawData2D.length); r++) { const rowStr = rawData2D[r].join(" ").toLowerCase(); if (rowStr.includes("date") || rowStr.includes("company") || rowStr.includes("security") || rowStr.includes("price") || rowStr.includes("vwap")) { headerRowIdx = r; break; } } const fileHeaders = rawData2D[headerRowIdx].map(h => String(h).toLowerCase().replace(/[^a-z0-9]/g, '')); const colMap = {}; targetHeaders.forEach(th => { let bestMatch = -1; for(let j = 0; j < fileHeaders.length; j++) { const h = fileHeaders[j]; if (!h) continue; if (th === 'DATE' && (h.includes('date') || h.includes('day'))) bestMatch = j; else if (th === 'ISIN' && h.includes('isin')) bestMatch = j; else if (th === 'COMPANY_NAME' && (h.includes('company') || h.includes('name') || h.includes('security') || h.includes('ticker') || h.includes('stock'))) bestMatch = j; else if (th === 'STATUS' && h.includes('status')) bestMatch = j; else if (th === 'VWAP' && (h.includes('vwap') || h.includes('close') || h.includes('last') || h.includes('ltp') || h.includes('price'))) bestMatch = j; else if (th === 'PREVIOUS' && (h.includes('prev') || h.includes('open'))) bestMatch = j; else if (th === 'Change' && (h.includes('change') || h.includes('diff'))) bestMatch = j; else if (th === 'VOLUME' && (h.includes('vol') || h.includes('shares') || h.includes('qty'))) bestMatch = j; else if (th === 'Turnover' && (h.includes('turnover') || h.includes('value'))) bestMatch = j; else if (th === 'INTRADAY_HIGH' && h.includes('high') && !h.includes('52')) bestMatch = j; else if (th === 'INTRADAY_LOW' && h.includes('low') && !h.includes('52')) bestMatch = j; else if (th === '52WK_HIGH' && h.includes('52') && h.includes('high')) bestMatch = j; else if (th === '52WK_LOW' && h.includes('52') && h.includes('low')) bestMatch = j; else if (th === 'BID' && h.includes('bid')) bestMatch = j; else if (th === 'ASK' && (h.includes('ask') || h.includes('offer'))) bestMatch = j; } colMap[th] = bestMatch; }); if (colMap["DATE"] === -1 || colMap["COMPANY_NAME"] === -1) continue; // Row Extraction & Cleansing for (let i = headerRowIdx + 1; i < rawData2D.length; i++) { const row = rawData2D[i]; if (row.length < 2 || !row[colMap["DATE"]]) continue; let rawDate = String(row[colMap["DATE"]]).trim(); let d = new Date(rawDate); if (isNaN(d.getTime())) continue; let sortableDate = Utilities.formatDate(d, "Africa/Nairobi", "yyyy-MM-dd"); uniqueDates.add(sortableDate); let companyName = colMap["COMPANY_NAME"] !== -1 ? String(row[colMap["COMPANY_NAME"]]).trim() : "UNKNOWN"; if (!companyName || companyName === "UNKNOWN" || companyName === "") continue; let isin = colMap["ISIN"] !== -1 ? String(row[colMap["ISIN"]]).trim() : ""; let status = colMap["STATUS"] !== -1 ? String(row[colMap["STATUS"]]).trim() : ""; if (!companyMemory[companyName]) { companyMemory[companyName] = { isin: isin, status: status }; } else { if (isin && !companyMemory[companyName].isin) companyMemory[companyName].isin = isin; if (status && !companyMemory[companyName].status) companyMemory[companyName].status = status; } if (!dataMap[sortableDate]) dataMap[sortableDate] = {}; const extractNum = (th) => { const idx = colMap[th]; if (idx === -1 || idx >= row.length) return ""; let val = String(row[idx]).trim(); if (val === "" || val === "-" || val.toLowerCase() === "n/a" || val.toLowerCase() === "null") return ""; let num = parseFloat(val.replace(/,/g, '').replace(/%/g, '').replace(/[A-Za-zKSh$]/g, '')); return isNaN(num) ? val : num; }; dataMap[sortableDate][companyName] = { vwap: extractNum("VWAP"), prev: extractNum("PREVIOUS"), change: extractNum("Change"), vol: extractNum("VOLUME"), turnover: extractNum("Turnover"), ihigh: extractNum("INTRADAY_HIGH"), ilow: extractNum("INTRADAY_LOW"), wkhigh: extractNum("52WK_HIGH"), wklow: extractNum("52WK_LOW"), bid: extractNum("BID"), ask: extractNum("ASK") }; } } // 2. Recursively process all sub-folders (Years -> Months) const subFolders = folder.getFolders(); while (subFolders.hasNext()) { processFolder(subFolders.next()); } } // Start the recursive engine on the root folder processFolder(rootFolder); const dateArray = Array.from(uniqueDates).sort((a, b) => new Date(b).getTime() - new Date(a).getTime()); const companyArray = Object.keys(companyMemory).sort((a, b) => a.localeCompare(b)); if (dateArray.length === 0 || companyArray.length === 0) { console.error(`❌ Data Extraction Failed: Processed ${processedFiles} files, but found 0 valid rows.`); return; } console.log(`⚙️ Building Cartesian Panel Matrix for ${dateArray.length} Dates and ${companyArray.length} Companies...`); const finalOutput = []; dateArray.forEach(dateISO => { const [yyyy, mm, dd] = dateISO.split("-"); const displayDate = `${dd}/${mm}/${yyyy}`; // Format as DD/MM/YYYY companyArray.forEach(company => { const meta = companyMemory[company]; const metrics = (dataMap[dateISO] && dataMap[dateISO][company]) ? dataMap[dateISO][company] : null; if (metrics) { finalOutput.push([ displayDate, meta.isin, company, meta.status, metrics.vwap, metrics.prev, metrics.change, metrics.vol, metrics.turnover, metrics.ihigh, metrics.ilow, metrics.wkhigh, metrics.wklow, metrics.bid, metrics.ask ]); } else { // Strict Cartesian Grid padding finalOutput.push([displayDate, meta.isin, company, meta.status, "", "", "", "", "", "", "", "", "", "", ""]); } }); }); console.log(`💾 Writing ${finalOutput.length.toLocaleString()} rows to Google Sheets in batches...`); // High-Performance Batch Writer const chunkSize = 5000; for (let i = 0; i < finalOutput.length; i += chunkSize) { const chunk = finalOutput.slice(i, i + chunkSize); sheet.getRange(i + 2, 1, chunk.length, chunk[0].length).setValues(chunk); SpreadsheetApp.flush(); } sheet.getRange(2, 5, sheet.getLastRow(), 11).setNumberFormat("#,##0.00"); sheet.autoResizeColumns(1, targetHeaders.length); console.log(`✅ SUCCESS: Time-Series Panel Built. Data is ready for analysis.`); }
